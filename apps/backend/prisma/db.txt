// Aqui no es modelo de better auth
model Noticia {
  id          Int      @id @default(autoincrement())
  titulo      String
  descripcion String
  publicado   DateTime @default(now())

  // Relacion con Better Auth User
  user   User   @relation(fields: [userId], references: [id])
  userId String

  recursos Recurso[]

  @@index([id])
}

model Recurso {
  id         Int      @id @default(autoincrement())
  resource   String
  fecha      DateTime @default(now())
  es_noticia Boolean  @default(false)

  noticia   Noticia? @relation(fields: [noticiaId], references: [id])
  noticiaId Int?

  user   User   @relation(fields: [userId], references: [id])
  userId String
}

model Ausencia {
  id      Int      @id @default(autoincrement())
  materia String
  creado  DateTime @default(now())

  // Docente que falta
  docente   User   @relation(name: "AusenciaDocente", fields: [docenteId], references: [id])
  docenteId String

  // Usuario que publica
  publicador   User   @relation(name: "AusenciaPublicador", fields: [publicadorId], references: [id])
  publicadorId String
}

model Eventos {
  id          Int      @id @default(autoincrement())
  nombre      String
  descripcion String
  fecha       DateTime
  creado      DateTime @default(now())

  user   User?   @relation(fields: [userId], references: [id])
  userId String?
}
